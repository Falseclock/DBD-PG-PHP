namespace DBD;
use DBI;

class <DRIVER_NAME> extends DBD{
	
    public function connect()
    {
        return new <DRIVER_NAME>Extend($this);
    }
}

class <DRIVER_NAME>Extend extends <DRIVER_NAME> implements DBI {
    private $query      = "";
    private $rows       = 0;
    private $affected   = 0;
 
    public function __construct($object, $statement = "")
    {
        foreach (get_object_vars($object) as $key => $value) {
            $this->$key = $value;
        }
        $this->query = $statement;
    }

    public function prepare($statement)
    {
        if ( ! isset($statement) or empty($statement))
            trigger_error("prepare failed: statement is not set or empty", E_USER_ERROR);

        return new <DRIVER_NAME>Extend($this, $statement);
    }
}
